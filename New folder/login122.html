function validateForm() { var username = document.getElementById("username").value; var password = document.getElementById("password").value; var errorMessages = ""; // Username validation var usernameRegex = /^(?=.*[A-Z])(?=.*[a-z])(?=.*\d)(?=.*[!@#$%^&*()])[A-Za-z\d!@#$%^&*()]{8,}$/; if (!usernameRegex.test(username)) { errorMessages += "Username should contain at least one uppercase letter, one lowercase letter, one special character, one number, and be at least 8 characters long.
"; } // Password validation var passwordRegex = /^(?=.*[A-Z])(?=.*[!@#$%^&*()])[A-Za-z\d!@#$%^&*()]{8,}$/; if (!passwordRegex.test(password)) { errorMessages += "Password should contain at least one uppercase letter, one special character, and be at least 8 characters long.
"; } // Display error messages or submit the form if (errorMessages !== "") { document.getElementById("errorMessages").innerHTML = '
' + errorMessages + '
'; return false; } else { // Submit the form or perform further actions return true; } }
